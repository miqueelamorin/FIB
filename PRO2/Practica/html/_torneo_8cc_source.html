43void Torneo::emparejar(BinTree<pair<int, string>>& a, int h, int H, int padre, int ninscritos, int potencia_ant)
44{
45 int potencia_act = potencia_ant * 2;
46 int b = (potencia_act + 1 - padre);
47 if (h > H or b > ninscritos) a = BinTree<pair<int, string>> (make_pair(padre,"0"));
48 else {
49 BinTree<pair<int, string>> l;
50 BinTree<pair<int, string>> r;
51 emparejar(l, h+1, H, padre, ninscritos, potencia_act);
52 emparejar(r, h+1, H, b, ninscritos, potencia_act);
53 a = BinTree<pair<int, string>> (make_pair(padre,"0"), l , r);

97bool Torneo::calcular_actualizar_estadisticas(string result, string jug1, string jug2, Cjt_Jugadores& conj)
98{
99 int sets1, sets2, juegos1, juegos2, partidos1, partidos2;
100 sets1 = sets2 = juegos1 = juegos2 = partidos1 = partidos2 = 0;
101
102 if (result != "1-0" and result != "0-1"){
103 for(int i = 0; i < result.length()-2; i+=4){
104 if(result[i] > result[i+2]) ++sets1;
105 else ++sets2;
106 juegos1 += result[i] - '0';
107 juegos2 += result[i+2] - '0';
108 }
109 if (sets1 > sets2) partidos1 = 1;
110 else partidos2 = 1;
111 }
112 else{
113 if (result[0] > result[2]) partidos1 = 1;
114 else partidos2 = 1;
115 }
116 conj.actualizar_estadisticas(jug1, 0, sets1, sets2, juegos1, juegos2, partidos1, partidos2);
117 conj.actualizar_estadisticas(jug2, 0, sets2, sets1, juegos2, juegos1, partidos2, partidos1);
118 return (partidos1 > partidos2);
119}
